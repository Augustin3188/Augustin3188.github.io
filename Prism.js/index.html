Công thức chuyển đổi phút sang giờ

                
                    function convertMinute(x = 60) {
                        let result = Math.floor(x / 60);
                        return result
                }
                    function convertMinuteresidual(x = 60) {
                        let result = x % 60;
                        return result
                }
                    let d = convertMinute(888);
                    let e = convertMinuteresidual(888)
                    console.log(`${d} giờ ${e} phút`)

Công thức tính BMI

                
                    function BMI(w = 50, h = 1.6) {
                        let result = w / (h ** 2)
                        return result
                    }
                    let h = BMI(66, 1.7)
                    console.log(`BMI = ${Math.round(h * 100) / 100}`)
